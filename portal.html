window.addEventListener('gfii:ready', async function (e) { const user = e.detail.user; if (!user) { location.href = '/login.html'; return; } try { await user.getUserData(); // 🔥 validates session const metadata = user.user_metadata || {}; if (!metadata.approved) { location.href = '/not-approved.html'; return; } if (!metadata.onboarding_complete) { location.href = '/onboarding.html'; return; } // ✅ SUCCESS document.getElementById('logout-btn').hidden = false; } catch (err) { console.error(err); location.href = '/login.html'; } });
